        var Count_Occ = (s) => 
{
   let n=s.length;
   let str=s.split("").sort().join("");
   let ans="";
   for(let i=0;i<n;i++) {
     let count=0;
     for(j=0;j<n;j++) {
       if (str[i]===str[j] && i>j) {
         break;
       }
       if(str[i]===str[j]) {
         count++;
       }
     }
     if (count>1) {
       ans+=str[i] + count;
     }
   }
   return ans;
};

Q6-Find the Winner-

        var Game_Winner = (S) => 
        {
          let countA = 0;
          let countD = 0;
          
          for(i=0;i<S.length;i++){
            if (S.charAt(i)=== 'A'){
              countA++;
            }
            else{
                countD++;
            }
          }
          
          if(countA>countD){
            return "Aditya";
          }
          
         else if(countA>countD){
            return "Danish";
          }
          
          else{
            return "Draw";
          }
        }
